SELECT ?lexp ?when ?location
FROM <http://data.open.ac.uk/context/led>
WHERE {
  ?lexp a led:ListeningExperience .
  ?lexp event:time ?time .
  ?time rdfs:label ?when .
  {
    ?time edtf:century edtfdata:Century.18 .
    ?time edtf:yearOfCentury/rdf:value ?yoc .
    FILTER (?yoc >= 37 && ?yoc <= 99)
  }
  UNION
  {
    ?time edtf:century edtfdata:Century.18 .
    ?time edtf:yearOfCentury/rdfs:label ?yoc .
    FILTER (?yoc IN ("40s"@en, "50s"@en, "60s"@en, "70s"@en, "80s"@en, "90s"@en))
  }
  UNION
  {
    ?time edtf:century edtfdata:Century.19 .
    ?time edtf:yearOfCentury/rdf:value ?yoc .
    FILTER (?yoc >= 00 && ?yoc <= 01)
  }
  OPTIONAL { ?lexp event:place ?location }
}
